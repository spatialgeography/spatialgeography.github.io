        },
        {

            text: "The 'Theory of Continental Drift' was proposed by:",
            options: ["Harry Hess", "Alfred Wegener", "Arthur Holmes", "J.T. Wilson"],
            correct: 1
        },
        {
            id: 10,
            text: "Which of the following is a sedimentary rock?",
            options: ["Granite", "Basalt", "Sandstone", "Marble"],
            correct: 2
        },
        {
            id: 11,
            text: "The 'Equilibrium theory of Tides' was given by:",
            options: ["Newton", "Laplace", "Harris", "Whewell"],
            correct: 0
        },
        {
            id: 12,
            text: "Which river is known as the 'Sorrow of Bihar'?",
            options: ["Ganga", "Kosi", "Gandak", "Damodar"],
            correct: 1
        },
        {
            id: 13,
            text: "The 'Isostatic theory' was first proposed by:",
            options: ["Dutton", "Pratt", "Airy", "Hayford"],
            correct: 0
        },
        {
            id: 14,
            text: "Which of the following is a cold ocean current?",
            options: ["Gulf Stream", "Kuroshio", "Canary", "Brazil"],
            correct: 2
        },
        {
            id: 15,
            text: "The term 'Megalopolis' was coined by:",
            options: ["Jean Gottmann", "Patrick Geddes", "Lewis Mumford", "C.A. Doxiadis"],
            correct: 0
        }
    ];

    let currentIdx = 0;
    let answers = new Array(questions.length).fill(null);
    let status = new Array(questions.length).fill(0); // 0:NotVisited, 1:NotAnswered, 2:Answered, 3:Marked, 4:AnsweredMarked
    let timeLeft = 30 * 60; // 30 minutes

    // Elements
    const qText = document.getElementById('question-text');
    const qNum = document.getElementById('question-number');
    const optContainer = document.getElementById('options-container');
    const palette = document.getElementById('question-palette');
    const timerEl = document.getElementById('timer');

    function init() {
        renderPalette();
        loadQuestion(0);
        startTimer();
    }

    function startTimer() {
        const interval = setInterval(() => {
            timeLeft--;
            const mins = Math.floor(timeLeft / 60);
            const secs = timeLeft % 60;
            timerEl.textContent = `Time Left: ${mins}:${secs < 10 ? '0' : ''}${secs}`;
            if (timeLeft <= 0) {
                clearInterval(interval);
                alert("Time is up! Submitting your test.");
                submitTest();
            }
        }, 1000);
    }

    function renderPalette() {
        palette.innerHTML = '';
        questions.forEach((q, i) => {
            const btn = document.createElement('button');
            btn.className = `palette-btn ${getStatusClass(status[i])} ${i === currentIdx ? 'current' : ''}`;
            btn.textContent = i + 1;
            btn.onclick = () => loadQuestion(i);
            palette.appendChild(btn);
        });
        updateLegendCounts();
    }

    function getStatusClass(s) {
        switch(s) {
            case 0: return 'not-visited';
            case 1: return 'not-answered';
            case 2: return 'answered';
            case 3: return 'marked';
            case 4: return 'answered-marked';
            default: return '';
        }
    }

    function updateLegendCounts() {
        const counts = [0,0,0,0,0];
        status.forEach(s => counts[s]++);
        document.querySelector('.badge-not-visited').textContent = counts[0];
        document.querySelector('.badge-not-answered').textContent = counts[1];
        document.querySelector('.badge-answered').textContent = counts[2];
        document.querySelector('.badge-marked').textContent = counts[3];
        document.querySelector('.badge-answered-marked').textContent = counts[4];
    }

    function loadQuestion(idx) {
        // Mark current as not answered if it was just visited
        if (status[currentIdx] === 0) status[currentIdx] = 1;

        currentIdx = idx;
        const q = questions[idx];
        qNum.textContent = `Question No. ${idx + 1}`;
        qText.textContent = q.text;

        optContainer.innerHTML = '';
        q.options.forEach((opt, i) => {
            const div = document.createElement('div');
            div.className = `option-item ${answers[idx] === i ? 'selected' : ''}`;
            div.innerHTML = `<input type="radio" name="option" ${answers[idx] === i ? 'checked' : ''}> <span>${opt}</span>`;
            div.onclick = () => selectOption(i);
            optContainer.appendChild(div);
        });

        renderPalette();
    }

    function selectOption(i) {
        answers[currentIdx] = i;
        const items = optContainer.querySelectorAll('.option-item');
        items.forEach((item, idx) => {
            item.classList.toggle('selected', idx === i);
            item.querySelector('input').checked = (idx === i);
        });
    }

    // Buttons
    document.getElementById('save-next-btn').onclick = () => {
        if (answers[currentIdx] !== null) {
            status[currentIdx] = 2;
        } else {
            status[currentIdx] = 1;
        }
        moveToNext();
    };

    document.getElementById('mark-review-btn').onclick = () => {
        if (answers[currentIdx] !== null) {
            status[currentIdx] = 4;
        } else {
            status[currentIdx] = 3;
        }
        moveToNext();
    };

    document.getElementById('clear-btn').onclick = () => {
        answers[currentIdx] = null;
        if (status[currentIdx] === 2) status[currentIdx] = 1;
        if (status[currentIdx] === 4) status[currentIdx] = 3;
        loadQuestion(currentIdx);
    };

    function moveToNext() {
        if (currentIdx < questions.length - 1) {
            loadQuestion(currentIdx + 1);
        } else {
            renderPalette();
            alert("You have reached the last question.");
        }
    }

    function submitTest() {
        let correct = 0;
        let incorrect = 0;
        let unanswered = 0;

        questions.forEach((q, i) => {
            // Only Answered (2) and Answered & Marked (4) are evaluated
            if (status[i] === 2 || status[i] === 4) {
                if (answers[i] === q.correct) {
                    correct++;
                } else {
                    incorrect++;
                }
            } else {
                unanswered++;
            }
        });

        document.getElementById('total-score').textContent = `${correct}/${questions.length}`;
        document.getElementById('correct-count').textContent = correct;
        document.getElementById('incorrect-count').textContent = incorrect;
        document.getElementById('unanswered-count').textContent = unanswered;
        document.getElementById('results-modal').style.display = 'flex';

        if (correct >= (questions.length * 0.7)) {
            confetti({
                particleCount: 150,
                spread: 70,
                origin: { y: 0.6 },
                colors: ['#2ecc71', '#3498db', '#9b59b6']
            });
        }
    }

    document.getElementById('submit-test-btn').onclick = () => {
        const answeredCount = status.filter(s => s === 2 || s === 4).length;
        if (confirm(`You have answered ${answeredCount} out of ${questions.length} questions. Are you sure you want to submit?`)) {
            submitTest();
        }
    };

    document.getElementById('close-results').onclick = () => {
        location.reload();
    };

    init();
});
</script>

Important Test Instructions

Please read the following instructions carefully before starting:

  1. Question Palette Symbols:
    • 1 Not Visited: You have not visited the question yet.
    • 2 Not Answered: You have visited but not answered the question.
    • 3 Answered: You have answered the question.
    • 4 Marked for Review: You have NOT answered the question, but have marked it for review.
    • 5 Answered & Marked: The question will be considered for evaluation even if you don’t revisit it.
  2. Navigation:
    • Click on Save & Next to save your answer and move to the next question.
    • Click on Mark for Review & Next if you want to flag a question for later.
    • You can directly jump to any question by clicking its number in the palette.
  3. Result:
    • Your final score will be displayed immediately after submission.
    • The test includes questions from Geomorphology, Climatology, Oceanography, and Indian Geography.

Good Luck with your preparation!
